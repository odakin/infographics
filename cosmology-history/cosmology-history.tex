% ---- Data box: 5-row aligned math environment ----
% Renders content as a single math array with 3 columns:
%   right-aligned label  |  centered relation  |  left-aligned value
% Use Japanese inside \text{...}; units (K, s, MeV...) also via \text{}.
% Rows without a relation symbol just leave the middle cell empty.
%
% Usage:
%   \databox{%
%     z              &           & \text{非常に大} \\
%     t              & \sim      & 10^{-36}\text{--}10^{-32}\,\text{s} \\
%     \text{過去}    & \approx   & 138\,\text{億年前} \\
%     ...
%   }
\newcommand{\databox}[1]{%
  \node[anchor=north west, font=\fontsize{7.4}{9.2}\selectfont, text=fgstrong, inner sep=0pt]
    at (-13, 5) {%
    $\renewcommand{\arraystretch}{1.0}\begin{array}{r@{\;}c@{\;}l}#1\end{array}$%
  };%
}
